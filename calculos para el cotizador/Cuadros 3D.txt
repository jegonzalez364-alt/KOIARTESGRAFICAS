function precioKoi3D(ancho_cm, alto_cm) {

    // costo por m² de cuadro 3D (doble madera + doble impresión)
    const costo3D_m2 = 106000;

    // cálculo de área
    const area = (ancho_cm / 100) * (alto_cm / 100);

    // multiplicador calibrado
    let multiplicador = 12.2 - (area * 9.8);

    // límites reales según tu catálogo
    if (multiplicador > 12.0) multiplicador = 12.0;
    if (multiplicador < 2.4)  multiplicador = 2.4;

    // precio base
    let precio = area * costo3D_m2 * multiplicador;

    // redondeo hacia ARRIBA Koi
    precio = Math.ceil(precio / 5000) * 5000;

    return precio;
}
