function precioKoiFraccionado(ancho_cm, alto_cm) {

    const area = (ancho_cm / 100) * (alto_cm / 100);
    const costo_m2 = 53180;

    // Fórmula suave y natural ajustada a tus precios reales
    // curva: multiplicador cae suavemente sin saltos
    let multiplicador = 4.55 - (area * 0.90);

    // LÍMITES para evitar que se pase o se hunda demasiado
    if (multiplicador > 4.25) multiplicador = 4.25;   // límite superior suave
    if (multiplicador < 2.44) multiplicador = 2.44;   // límite inferior suave

    // precio bruto
    let precio = area * costo_m2 * multiplicador;

    // REDONDEO hacia ARRIBA
    precio = Math.ceil(precio / 5000) * 5000;

    return precio;
}
